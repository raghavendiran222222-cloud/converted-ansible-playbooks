# Run this on servers
Set-Item WSMan:\\localhost\client\TrustedHosts -Value * -Concatenate -Force


# Join servers to domain
$Credential = get-credential

$Add_Computer_Splatter = @{
    DomainName  = "world.fluidtechnology.net"
    OUPath      = "OU=AzureAutoDeployed,OU=Servers,OU=AZ-USE2,OU=_AccountObjects,DC=world,DC=fluidtechnology,DC=net"
    Credential  = $Credential
    ErrorAction = 'Stop'
    Force       = $true
    #WhatIf       = $true #Change
}

Add-Computer @Add_Computer_Splatter
